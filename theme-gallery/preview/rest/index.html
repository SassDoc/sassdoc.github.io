clearfix

Source

@mixin clearfix($extend: true) { 
  @if $extend {
    @extend %clearfix;
  }

  @else {
    &:after {
      content: "";
      display: table;
      clear: both;
    }
  }
}

Description

Float clearing helper.

visually-hidden

Source

@mixin visually-hidden($extend: true) { 
  @if $extend {
    @extend %visually-hidden;
  }

  @else {
    @include size(1px);
    position: absolute;
    padding: 0;
    margin: -1px;
    overflow: hidden;
    clip: rect(0 0 0 0);
    border: 0;
  }
}

Description

Accessibility hiding helper.

  @if $extend {
